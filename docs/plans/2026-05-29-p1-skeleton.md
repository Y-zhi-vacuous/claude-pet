# P1: 骨架阶段 — 实现计划

> **For agentic workers:** 使用 superpowers:subagent-driven-development 或 superpowers:executing-plans 按任务逐步实施。步骤使用 checkbox (`- [ ]`) 语法跟踪。

**Goal:** 让一个透明 WPF 窗口出现在桌面上，里面有一个彩色方块占位狗，可拖拽移动，有系统托盘图标，配置可从 JSON 加载。

**Architecture:** WPF 独立桌面应用，`WindowStyle=None` + `AllowsTransparency=True` 实现透明无边框窗口，`Topmost=true` 浮于桌面。动画系统通过 `IAnimationPlayer` 接口抽象，初始用彩色方块旋转模拟。所有配置放在 `config.json`。

**Tech Stack:** .NET 8, C# 12, WPF, Newtonsoft.Json

**Note:** 本项目不是 git 仓库（全新项目），跳过 commit 步骤，用文件写入确认代替。

```csharp
/// <summary>
```

```csharp
    /// <summary>获取当前动画素材的逻辑尺寸（宽, 高），单位像素。</summary>
    (int Width, int Height) GetSize();

    /// <summary>当一段非循环动画播放完成时触发。参数为播放结束的状态名。</summary>
    event Action<AnimationState>? AnimationEnded;
```

```csharp
using ClaudePet.Models;

namespace ClaudePet.Animation;

/// <summary>
/// 动画资源目录：记录每个 AnimationState 对应哪些帧文件。
/// P1 阶段全部指向占位的彩色方块，后续替换为实际 PNG 路径即可。
/// </summary>
public static class AnimationCatalog
{
    /// <summary>
    /// 获取指定动画状态的所有帧。
    /// P1 阶段：每个状态返回 4 个占位帧（纯色方块），
    /// 颜色随帧变化以肉眼可见的方式验证动画在播放。
    /// </summary>
    public static List<AnimationFrame> GetFrames(AnimationState state)
    {
        var colors = state switch
        {
            AnimationState.Idle      => new[] { "#FFF8E1", "#FFF3CD", "#FFE69C", "#FFF8E1" },
            AnimationState.WalkLeft  => new[] { "#D4E6F1", "#A9CCE3", "#7FB3D8", "#D4E6F1" },
            AnimationState.WalkRight => new[] { "#D5F5E3", "#A9DFBF", "#7DCEA0", "#D5F5E3" },
            AnimationState.WalkUp    => new[] { "#FADBD8", "#F5B7B1", "#F1948A", "#FADBD8" },
            AnimationState.WalkDown  => new[] { "#E8DAEF", "#D2B4DE", "#BB8FCE", "#E8DAEF" },
            AnimationState.SitSleep  => new[] { "#D5DBDB", "#ABB2B9", "#7F8C8D", "#D5DBDB" },
            AnimationState.Play      => new[] { "#FCF3CF", "#F9E79F", "#F7DC6F", "#FCF3CF" },
            AnimationState.Think     => new[] { "#FDEBD0", "#FAD7A1", "#F8C471", "#FDEBD0" },
            AnimationState.Talk      => new[] { "#FDEDEC", "#FADBD8", "#F1948A", "#FDEDEC" },
            AnimationState.Happy     => new[] { "#D4EFDF", "#A9DFBF", "#7DCEA0", "#D4EFDF" },
            AnimationState.Eat       => new[] { "#F2F3F4", "#D5D8DC", "#ABB2B9", "#F2F3F4" },
            _                        => new[] { "#CCCCCC", "#AAAAAA", "#888888", "#CCCCCC" }
        };

        return colors.Select((hex, i) => new AnimationFrame
        {
            ImagePath = hex, // P1 占位：直接用颜色值代替图片路径
            DurationMs = 200
        }).ToList();
    }
}
```

**Step 1: 创建 SpriteSheetPlayer.cs**

```csharp
using System.Windows;
using System.Windows.Controls;
using System.Windows.Media;
using System.Windows.Shapes;
using ClaudePet.Models;

namespace ClaudePet.Animation;

/// <summary>
/// 基于 WPF 控件 + DispatcherTimer 的精灵动画播放器（P1 占位版）。
/// 用彩色矩形代替 PNG 帧，后续替换为 Image 控件 + 帧文件读取。
/// </summary>
public class SpriteSheetPlayer : IAnimationPlayer, IDisposable
{
    private readonly Canvas _canvas;
    private readonly Rectangle _rect;
    private readonly System.Windows.Threading.DispatcherTimer _timer;
    private List<AnimationFrame> _frames = new();
    private int _frameIndex;
    private AnimationState _currentState;
    private bool _disposed;

    public event Action<AnimationState>? AnimationEnded;

    public SpriteSheetPlayer(Canvas canvas)
    {
        _canvas = canvas;
        _rect = new Rectangle
        {
            Width = 140,
            Height = 140,
            RadiusX = 20,
            RadiusY = 20,
            Fill = new SolidColorBrush(Colors.Gray)
        };
        _canvas.Children.Add(_rect);
        Canvas.SetLeft(_rect, 0);
        Canvas.SetTop(_rect, 0);

        _timer = new System.Windows.Threading.DispatcherTimer
        {
            Interval = TimeSpan.FromMilliseconds(200)
        };
        _timer.Tick += OnTick;

        Play(AnimationState.Idle);
    }

    public void Play(AnimationState state)
    {
        if (_currentState == state) return;

        _currentState = state;
        _frames = AnimationCatalog.GetFrames(state);
        _frameIndex = 0;
        RenderFrame();
        _timer.Start();
    }

    public void Stop()
    {
        _timer.Stop();
    }

    public void SetRenderPosition(double x, double y)
    {
        Canvas.SetLeft(_rect, x);
        Canvas.SetTop(_rect, y);
    }

    public (int Width, int Height) GetSize() => (140, 140);

    private void OnTick(object? sender, EventArgs e)
    {
        _frameIndex++;
        if (_frameIndex >= _frames.Count)
        {
            // 对于循环动画（Idle、Walk 系列），回到第 0 帧
            if (IsLooping(_currentState))
            {
                _frameIndex = 0;
            }
            else
            {
                _timer.Stop();
                AnimationEnded?.Invoke(_currentState);
                return;
            }
        }
        RenderFrame();
    }

    private void RenderFrame()
    {
        if (_frameIndex >= _frames.Count) return;
        var frame = _frames[_frameIndex];
        try
        {
            var color = (Color)ColorConverter.ConvertFromString(frame.ImagePath);
            _rect.Fill = new SolidColorBrush(color);
        }
        catch
        {
            _rect.Fill = new SolidColorBrush(Colors.Gray);
        }
    }

    private static bool IsLooping(AnimationState state) => state switch
    {
        AnimationState.Idle or
        AnimationState.WalkLeft or
        AnimationState.WalkRight or
        AnimationState.WalkUp or
        AnimationState.WalkDown => true,
        _ => false
    };

    public void Dispose()
    {
        if (_disposed) return;
        _disposed = true;
        _timer.Stop();
        _canvas.Children.Remove(_rect);
    }
}
```

```csharp
using Newtonsoft.Json;
using ClaudePet.Models;

namespace ClaudePet.Utils;

/// <summary>
/// 配置文件读写。配置存储在应用目录下的 config.json。
/// </summary>
public class ConfigStore
{
    private readonly string _path;

    public ConfigStore(string baseDir)
    {
        _path = Path.Combine(baseDir, "config.json");
    }

    public PetConfig Load()
    {
        if (!File.Exists(_path))
        {
            var defaults = new PetConfig();
            Save(defaults);
            return defaults;
        }

        var json = File.ReadAllText(_path);
        return JsonConvert.DeserializeObject<PetConfig>(json) ?? new PetConfig();
    }

    public void Save(PetConfig config)
    {
        var json = JsonConvert.SerializeObject(config, Formatting.Indented);
        File.WriteAllText(_path, json);
    }
}
```

```csharp
using System.Windows;
using System.Windows.Input;

namespace ClaudePet.Utils;

/// <summary>
/// 让无边框窗口可以通过鼠标左键拖拽移动。
/// 在 PetWindow 的 MouseLeftButtonDown/MouseMove/MouseLeftButtonUp 事件中使用。
/// </summary>
public class MouseDragHelper
{
    private readonly Window _window;
    private bool _isDragging;
    private Point _dragStartPoint;

    public MouseDragHelper(Window window)
    {
        _window = window;
    }

    public void OnMouseDown(Point mousePos)
    {
        _isDragging = true;
        _dragStartPoint = mousePos;
        _window.CaptureMouse();
    }

    public void OnMouseMove(Point mousePos)
    {
        if (!_isDragging) return;

        var deltaX = mousePos.X - _dragStartPoint.X;
        var deltaY = mousePos.Y - _dragStartPoint.Y;

        _window.Left += deltaX;
        _window.Top += deltaY;

        // 不需要更新 _dragStartPoint，因为 window 位置已经变了，
        // 但鼠标在屏幕上的绝对位置也变了，差值保持不变，用相对移动即可。
        // 使用 DragMove() 替代方案会更简单，但会在某些情况下产生跳动。
    }

    public void OnMouseUp()
    {
        _isDragging = false;
        _window.ReleaseMouseCapture();
    }
}
```

```csharp
using System.Drawing;
using System.Windows;
using ClaudePet.Utils;

namespace ClaudePet.UI;

/// <summary>
/// 系统托盘图标 + 右键菜单管理。
/// 这里用 Application.Current.Exit() 而不是 Kill 进程。
/// </summary>
public class TrayManager : IDisposable
{
    private readonly NotifyIcon _notifyIcon;
    private readonly ContextMenuStrip _contextMenu;

    public TrayManager()
    {
        _contextMenu = new ContextMenuStrip();
        _contextMenu.Items.Add("切换会话", null, (_, _) => { /* P2 实现 */ });
        _contextMenu.Items.Add("暂停动画", null, OnPauseAnimation);
        _contextMenu.Items.Add("-");
        _contextMenu.Items.Add("设置", null, (_, _) => { /* P5 实现 */ });
        _contextMenu.Items.Add("退出", null, OnExit);

        _notifyIcon = new NotifyIcon
        {
            Text = $"Claude 小狗桌宠",
            ContextMenuStrip = _contextMenu,
            Visible = true
        };

        // 用一个简单的狗爪印图标占位，后面替换为 .ico 文件
        using var bitmap = new Bitmap(32, 32);
        using var g = Graphics.FromImage(bitmap);
        g.Clear(System.Drawing.Color.Transparent);
        using var brush = new SolidBrush(System.Drawing.Color.DarkOrange);
        g.FillEllipse(brush, 2, 2, 28, 28);
        var iconHandle = bitmap.GetHicon();
        _notifyIcon.Icon = Icon.FromHandle(iconHandle);
    }

    private void OnPauseAnimation(object? sender, EventArgs e)
    {
        var item = (ToolStripMenuItem?)_contextMenu.Items[1];
        if (item == null) return;
        // 通过 Application 查找 PetWindow 并切换暂停
        var window = Application.Current.Windows.OfType<PetWindow>().FirstOrDefault();
        if (window != null)
        {
            window.AnimationPaused = !window.AnimationPaused;
            item.Text = window.AnimationPaused ? "恢复动画" : "暂停动画";
        }
    }

    private void OnExit(object? sender, EventArgs e)
    {
        _notifyIcon.Visible = false;
        Application.Current.Shutdown();
    }

    public void Dispose()
    {
        _notifyIcon.Visible = false;
        _notifyIcon.Dispose();
        _contextMenu.Dispose();
    }
}
```

```csharp
using System.Windows;

using ClaudePet.Animation;
using ClaudePet.Models;

namespace ClaudePet.UI;

public partial class PetWindow : Window
{
    private readonly ConfigStore _config;
    private readonly MouseDragHelper _dragHelper;
    private IAnimationPlayer? _animationPlayer;
    private bool _animationPaused;

    public bool AnimationPaused
    {
        get => _animationPaused;
        set
        {
            _animationPaused = value;
            if (value)
                _animationPlayer?.Stop();
            else
                _animationPlayer?.Play(AnimationState.Idle);
        }
    }

    public PetWindow(ConfigStore config)
    {
        _config = config;
        _dragHelper = new MouseDragHelper(this);
        InitializeComponent();
    }

    private void Window_Loaded(object sender, RoutedEventArgs e)
    {
        var petConfig = _config.Load();
        AnimationPaused = petConfig.AnimationPaused;

        // 初始化动画播放器（P1：彩色方块占位）
        _animationPlayer = new SpriteSheetPlayer(PetCanvas);

        // 设置初始位置
        if (petConfig.WindowLeft < 0 || petConfig.WindowTop < 0)
        {
            // 自动定位到屏幕右下角
            var screen = System.Windows.Forms.Screen.PrimaryScreen;
            if (screen != null)
            {
                Left = screen.WorkingArea.Right - Width - 20;
                Top = screen.WorkingArea.Bottom - Height - 60;
            }
        }
        else
        {
            Left = petConfig.WindowLeft;
            Top = petConfig.WindowTop;
        }
    }

    {
        // 保存当前位置
        var cfg = _config.Load();
        cfg.WindowLeft = Left;

    {
        // 右键由 TrayManager 处理，这里不需要额外操作。
        // Blank 占位，保持事件链完整性。
    }
```

```csharp
        // 获取项目根目录（config.json 所在位置）
        var baseDir = AppDomain.CurrentDomain.BaseDirectory;
        // 开发模式下 baseDir 在 bin/Debug/net8.0-windows/，需要上溯到项目根目录
        while (!File.Exists(Path.Combine(baseDir, "config.json")) && baseDir.Length > 3)
        {
            baseDir = Path.GetDirectoryName(baseDir)!;
        }

        var config = new ConfigStore(baseDir);

        _trayManager = new TrayManager();
        _petWindow = new PetWindow(config);
        _petWindow.Show();
    }

    private void Application_Exit(object sender, ExitEventArgs e)
    {
        _trayManager?.Dispose();
    }
}
```
